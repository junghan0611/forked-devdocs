요약

applets document 내의 정렬된 애플릿 목록을 리턴합니다.

문법

js

nodeList = document.applets

예제

js

// ( 여러분이 원하는 두 번째 애플릿을 알고 있을 때 )
my_java_app = document.applets[1];
